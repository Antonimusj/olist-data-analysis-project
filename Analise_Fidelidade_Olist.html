estado_query = '''
SELECT
c.customer_state AS estado,
ROUND(SUM(p.payment_value), 2) AS faturamento_total
FROM cliente c
JOIN pedido o ON c.customerid = o.customerid
JOIN pagamento p ON o.orderid = p.orderid
GROUP BY estado
ORDER BY faturamento_total DESC
LIMIT 10
'''

estado_df = pd.read_sql(estado_query, con=engine)

ticket_medio_query = '''
SELECT

JOIN pagamento p ON o.orderid = p.orderid
GROUP BY estado
ORDER BY ticket_medio DESC
'''

pagamento_query = '''
SELECT p.payment_type AS metodo_pagamento,
COUNT(*) AS quantidade_uso,
ROUND(SUM(payment_value),2) AS faturamento_total,
ROUND(AVG(p.payment_value),2) AS ticket_medio
FROM pagamento p
GROUP BY metodo_pagamento
ORDER BY quantidade_uso DESC
'''

metodo_pagamento_df = pd.read_sql(pagamento_query, con=engine)

parcelamento_query = '''
SELECT
c.customer_state AS estado,
ROUND(AVG(p.payment_installments),2) AS media_parcelas,
MAX(p.payment_installments) AS parcelas_maximas
FROM cliente c
JOIN pedido o ON c.customerid = o.customerid
JOIN pagamento p ON o.orderid = p.orderid
WHERE p.payment_type = "credit_card"
GROUP BY estado
ORDER BY media_parcelas DESC
'''

parcelas_df = pd.read_sql(parcelamento_query, con=engine)

recorrencia_query = '''
SELECT
c.customer_uniqueid AS cliente,
COUNT(DISTINCT o.orderid) AS total_pedidos,
ROUND(SUM(p.payment_value),2) AS total_gasto,
ROUND(AVG(p.payment_value),2) AS media_valor
FROM cliente c
JOIN pedido o ON c.customerid = o.customerid
JOIN pagamento p ON o.orderid = p.orderid
GROUP BY cliente
HAVING total_pedidos > 1
ORDER BY total_pedidos DESC
'''

recorrentes_df = pd.read_sql(recorrencia_query, con=engine)

Faturamento por Estado: Observa-se a dominncia de SP e da regio Sudeste/Sul em volume total. Isso indica onde est a maior concentrao de clientes e o maior fluxo de caixa da operao.

Ticket Mdio (Oportunidade Norte/Nordeste): Diferente do faturamento, os maiores tickets mdios esto concentrados nas regies Norte e Nordeste. Isso sugere um perfil de compra de maior valor agregado nessas regies, possivelmente influenciado pelo custo do frete ou pela escolha de produtos premium. O Mliuz pode explorar campanhas de "Cashback fixo para frete" ou benefcios especficos para compras de alto valor nessas reas.

Mtodos de Pagamento: A predominncia do Carto de Crdito o cenrio ideal para o modelo de negcio do Mliuz, facilitando a integrao de benefcios diretamente na fatura ou via parcerias bancrias.

ndice de Fidelidade: Os 3,12% representam a nossa maior oportunidade. Converter uma pequena frao da base de "compra nica" em clientes recorrentes geraria um aumento massivo de receita sem os custos de aquisio de novos usurios.

Concluso Estratgica: A anlise demonstra que a base possui um alto volume de usurios nicos, porm com um ndice de reteno de 3,12%. A mdia de pedidos do grupo fiel (2,12) indica que a maioria dos clientes que retornam o fazem apenas para uma segunda compra. O foco estratgico deve ser a implementao de gatilhos de recorrncia para mover os clientes da primeira para a segunda e terceira compras, onde o custo de aquisio (CAC) diludo e o valor vitalcio (LTV) do cliente se torna exponencial para a plataforma.
